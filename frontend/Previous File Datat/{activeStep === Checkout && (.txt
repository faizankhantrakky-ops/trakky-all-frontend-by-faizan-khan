                      // Validate that the amount doesn't exceed total
                      if (roundedAmount > totalAmount) {
                        return;
                      }

                      const updatedDetails = [...splitPaymentDetails];
                      updatedDetails[index].amount = roundedAmount;

                      // Auto-fill remaining amount to other modes if this is not the last field
                      const filledAmount = updatedDetails.reduce(
                        (sum, d) => sum + d.amount,
                        0
                      );
                      const remainingAmount =
                        totalAmount - filledAmount;

                      // If there's exactly one remaining unfilled field, auto-fill it
                      const unfilledFields = updatedDetails.filter(
                        (d, i) => i !== index && d.amount === 0
                      );

                      if (
                        unfilledFields.length === 1 &&
                        remainingAmount > 0
                      ) {
                        const remainingIndex = updatedDetails.findIndex(
                          (d, i) => i !== index && d.amount === 0
                        );
                        if (remainingIndex !== -1) {
                          // For wallet, ensure we don't exceed balance
                          const remainingDetail =
                            updatedDetails[remainingIndex];
                          if (remainingDetail.mode === "wallet") {
                            updatedDetails[remainingIndex].amount =
                              Math.round(
                                Math.min(
                                  remainingAmount,
                                  currentWalletBalance
                                ) * 100
                              ) / 100;
                          } else {
                            updatedDetails[remainingIndex].amount =
                              Math.round(remainingAmount * 100) / 100;
                          }
                        }
                      }

                      setSplitPaymentDetails(updatedDetails);

                      // Calculate total amount paid
                      const totalPaid = updatedDetails.reduce(
                        (sum, d) => sum + d.amount,
                        0
                      );
                      setAmountPaid(Math.round(totalPaid * 100) / 100);

                      // Update payment status based on total
                      if (totalPaid >= totalAmount) {
                        setPaymentStatus("paid");
                      } else if (
                        totalPaid > 0 &&
                        totalPaid < totalAmount
                      ) {
                        setPaymentStatus("partial");
                      } else if (totalPaid === 0 && totalAmount > 0) {
                        setPaymentStatus("unpaid");
                      }
                    }}
                    onBlur={(e) => {
                      const newAmount = parseFloat(e.target.value) || 0;
                      const roundedAmount = Math.round(newAmount * 100) / 100;
                      const updatedDetails = [...splitPaymentDetails];
                      updatedDetails[index].amount = roundedAmount;
                      setSplitPaymentDetails(updatedDetails);
                    }}
                    inputProps={{
                      min: 0,
                      max: isWalletMode
                        ? Math.min(
                            Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100,
                            currentWalletBalance
                          )
                        : Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100,
                      step: 0.01,
                    }}
                    error={
                      isWalletMode
                        ? detail.amount > currentWalletBalance
                        : false
                    }
                    helperText={
                      isWalletMode &&
                      detail.amount > currentWalletBalance
                        ? `Exceeds wallet balance (₹${Math.round(currentWalletBalance * 100) / 100})`
                        : ""
                    }
                  />
                </div>
                <Typography
                  variant="body2"
                  className="text-gray-600 capitalize"
                >
                  {detail.mode}
                  {isWalletMode &&
                    ` (Balance: ₹${Math.round(currentWalletBalance * 100) / 100})`}
                </Typography>

                {/* Auto-fill buttons */}
                {index === splitPaymentDetails.length - 1 &&
                  splitPaymentDetails.length > 1 && (
                    <Button
                      size="small"
                      variant="outlined"
                      onClick={() => {
                        const totalAmount = Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100;
                        const currentTotal = splitPaymentDetails.reduce(
                          (sum, d) => sum + d.amount,
                          0
                        );
                        const remainingAmount =
                          totalAmount - currentTotal;

                        if (remainingAmount > 0) {
                          const updatedDetails = [
                            ...splitPaymentDetails,
                          ];
                          // For wallet mode, don't exceed balance
                          if (isWalletMode) {
                            updatedDetails[index].amount = Math.round(
                              Math.min(
                                updatedDetails[index].amount +
                                  remainingAmount,
                                currentWalletBalance
                              ) * 100
                            ) / 100;
                          } else {
                            updatedDetails[index].amount = Math.round(
                              (
                                updatedDetails[index].amount +
                                remainingAmount
                              ) * 100
                            ) / 100;
                          }
                          setSplitPaymentDetails(updatedDetails);

                          const totalPaid = updatedDetails.reduce(
                            (sum, d) => sum + d.amount,
                            0
                          );
                          setAmountPaid(Math.round(totalPaid * 100) / 100);
                          if (totalPaid >= totalAmount) {
                            setPaymentStatus("paid");
                          }
                        }
                      }}
                      disabled={
                        splitPaymentDetails.reduce(
                          (sum, d) => sum + d.amount,
                          0
                        ) >= Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100 ||
                        (isWalletMode && currentWalletBalance <= 0)
                      }
                    >
                      Auto-fill Remaining
                    </Button>
                  )}
              </div>
            );
          })}

          {/* Enhanced validation summary with wallet info */}
          {splitPaymentDetails.length > 0 && (
            <div className="mt-2 p-2 bg-gray-50 rounded">
              <div className="flex justify-between text-sm">
                <span>Total Split Amount:</span>
                <span className="font-semibold">
                  ₹
                  {Math.round(
                    splitPaymentDetails.reduce((sum, d) => sum + d.amount, 0) * 100
                  ) / 100}
                </span>
              </div>
              <div className="flex justify-between text-sm">
                <span>Final Amount:</span>
                <span className="font-semibold">
                  ₹{Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100}
                </span>
              </div>
              <div className="flex justify-between text-sm">
                <span>Remaining Amount:</span>
                <span className="font-semibold">
                  ₹
                  {Math.round((
                    Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100 -
                    splitPaymentDetails.reduce(
                      (sum, d) => sum + d.amount,
                      0
                    )
                  ) * 100) / 100}
                </span>
              </div>

              {/* Wallet balance info */}
              {isWalletApplied && selectedWallet && (
                <div className="flex justify-between text-sm mt-1">
                  <span>Wallet Balance:</span>
                  <span className="font-semibold text-green-600">
                    ₹
                    {Math.round(
                      (wallets.find((w) => w.id === selectedWallet)
                        ?.remaining_price_benefits || 0) * 100
                    ) / 100}
                  </span>
                </div>
              )}

              {splitPaymentDetails.reduce(
                (sum, d) => sum + d.amount,
                0
              ) > Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100 && (
                <Alert severity="error" className="mt-1">
                  Total split amount exceeds final amount! Please adjust
                  the amounts.
                </Alert>
              )}

              {splitPaymentDetails.reduce(
                (sum, d) => sum + d.amount,
                0
              ) < Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100 && (
                <Alert severity="info" className="mt-1">
                  Amount not fully allocated. Remaining amount will be
                  added to due.
                </Alert>
              )}

              {Math.abs(
                splitPaymentDetails.reduce(
                  (sum, d) => sum + d.amount,
                  0
                ) - Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100
              ) < 0.01 && (
                <Alert severity="success" className="mt-1">
                  Amount fully allocated!
                </Alert>
              )}
            </div>
          )}
        </div>

        {/* Quick Action Buttons with wallet consideration */}
        {splitPaymentDetails.length > 1 && (
          <div className="mt-3 flex gap-2">
            <Button
              size="small"
              variant="outlined"
              onClick={() => {
                // Equal distribution considering wallet limits
                const totalAmount = Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100;
                const equalAmount =
                  totalAmount / splitPaymentDetails.length;
                const updatedDetails = splitPaymentDetails.map(
                  (detail) => {
                    if (detail.mode === "wallet") {
                      const walletBalance =
                        wallets.find((w) => w.id === selectedWallet)
                          ?.remaining_price_benefits || 0;
                      return {
                        ...detail,
                        amount: Math.round(
                          Math.min(equalAmount, walletBalance) * 100
                        ) / 100,
                      };
                    }
                    return {
                      ...detail,
                      amount: Math.round(equalAmount * 100) / 100,
                    };
                  }
                );

                // Adjust for rounding errors and wallet limits
                const totalDistributed = updatedDetails.reduce(
                  (sum, d) => sum + d.amount,
                  0
                );
                if (totalDistributed < totalAmount) {
                  // Distribute remaining to non-wallet modes
                  const remaining = totalAmount - totalDistributed;
                  const nonWalletModes = updatedDetails.filter(
                    (d) => d.mode !== "wallet"
                  );
                  if (nonWalletModes.length > 0) {
                    const perMode = remaining / nonWalletModes.length;
                    nonWalletModes.forEach((_, idx) => {
                      const detailIndex = updatedDetails.findIndex(
                        (d) => d.mode !== "wallet"
                      );
                      if (detailIndex !== -1) {
                        updatedDetails[detailIndex].amount = Math.round(
                          (
                            updatedDetails[detailIndex].amount + perMode
                          ) * 100
                        ) / 100;
                      }
                    });
                  }
                }

                setSplitPaymentDetails(updatedDetails);
                const finalTotal = updatedDetails.reduce(
                  (sum, d) => sum + d.amount,
                  0
                );
                setAmountPaid(Math.round(finalTotal * 100) / 100);
                setPaymentStatus(
                  finalTotal >= totalAmount ? "paid" : "partial"
                );
              }}
            >
              Distribute Equally
            </Button>

            <Button
              size="small"
              variant="outlined"
              onClick={() => {
                // Use maximum available from wallet first
                if (isWalletApplied && selectedWallet) {
                  const walletBalance =
                    wallets.find((w) => w.id === selectedWallet)
                      ?.remaining_price_benefits || 0;
                  const totalAmount =
                    Math.round((finalTotalAppointmentAmountAfterTaxDiscount + finalTotalProductSellAmountAfterDiscountTax) * 100) / 100;
                  const walletAmount = Math.min(
                    walletBalance,
                    totalAmount
                  );

                  const updatedDetails = splitPaymentDetails.map(
                    (detail) => ({
                      ...detail,
                      amount:
                        detail.mode === "wallet" ? Math.round(walletAmount * 100) / 100 : 0,
                    })
                  );

                  setSplitPaymentDetails(updatedDetails);
                  setAmountPaid(Math.round(walletAmount * 100) / 100);
                  setPaymentStatus(
                    walletAmount >= totalAmount ? "paid" : "partial"
                  );
                } else {
                  // Clear all amounts
                  const updatedDetails = splitPaymentDetails.map(
                    (detail) => ({
                      ...detail,
                      amount: 0,
                    })
                  );
                  setSplitPaymentDetails(updatedDetails);
                  setAmountPaid(0);
                  setPaymentStatus("unpaid");
                }
              }}
            >
              {isWalletApplied ? "Use Max Wallet" : "Clear All"}
            </Button>
          </div>
        )}
      </div>
    )}

    {/* Wallet information alert */}
    {isWalletApplied && selectedWallet && (
      <div className="w-full col-span-2">
        <Alert
          severity={
            wallets.find((w) => w.id === selectedWallet)
              ?.remaining_price_benefits === 0
              ? "error"
              : "info"
          }
          className="mt-2"
        >
          {wallets.find((w) => w.id === selectedWallet)
            ?.remaining_price_benefits === 0
            ? "Your wallet balance is 0 rupees. Please recharge your wallet to make payment."
            : `Available wallet balance: ₹${Math.round(
                (wallets.find((w) => w.id === selectedWallet)
                  ?.remaining_price_benefits || 0) * 100
              ) / 100}. ${
                splitPaymentMode
                  ? `Wallet will be used for partial payment as configured above.`
                  : `You can use split payment to use wallet partially with other methods.`
              }`}
        </Alert>
      </div>
    )}
  </>
)}
